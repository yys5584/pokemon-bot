작성일: 2026-03-08 | 기반: feature_proposals.md #2, #6

개요

채팅방 레벨 시스템 + 채널 네트워크를 구현하여:

  • 채널 운영자에게 봇 유지 인센티브 제공
  • 유저 활동량 증가 (레벨업 동기)
  • 채팅방 간 유저 풀 확대 (크로스 프로모션)

기존 인프라 활용:

  • chat_rooms.spawn_multiplier → 레벨별 스폰 보너스에 연결
  • chat_rooms.invite_link → 채널 네트워크 디렉토리에 활용
  • chat_activity 테이블 → CXP 보조 지표로 활용
  • spawn_service.calculate_daily_spawns() → 레벨 보너스 주입 지점

-- chat_rooms 테이블에 컬럼 추가 (새 테이블 X, 기존 확장) ALTER TABLE chat_rooms ADD COLUMN cxp INTEGER NOT NULL DEFAULT 0; ALTER TABLE chat_rooms ADD COLUMN chat_level INTEGER NOT NULL DEFAULT 1; ALTER TABLE chat_rooms ADD COLUMN cxp_today INTEGER NOT NULL DEFAULT 0; -- 일일 CXP 기록 (분석용, 선택) CREATE TABLE IF NOT EXISTS chat_cxp_log ( id SERIAL PRIMARY KEY, chat_id BIGINT REFERENCES chat_rooms(chat_id), action TEXT NOT NULL, -- 'catch', 'battle', 'trade' user_id BIGINT, amount INTEGER NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX idx_cxp_log_chat ON chat_cxp_log(chat_id, created_at);

# ─── 채팅방 레벨 ───────────────────────────────── CHAT_LEVEL_THRESHOLDS = [ # (level, required_cxp, spawn_bonus, rarity_boosts) (1, 0, 0, {}), (2, 100, 1, {}), (3, 300, 1, {"epic": 1.10}), (4, 600, 2, {"epic": 1.10, "legendary": 1.05}), (5, 1000, 2, {"epic": 1.10, "legendary": 1.05}), (6, 1500, 2, {"epic": 1.10, "legendary": 1.05}), (7, 2000, 2, {"epic": 1.10, "legendary": 1.05, "shiny": 1.03}), (8, 3000, 3, {"epic": 1.15, "legendary": 1.08, "shiny": 1.05}), (9, 5000, 3, {"epic": 1.15, "legendary": 1.10, "shiny": 1.05}), (10, 8000, 4, {"epic": 1.20, "legendary": 1.15, "shiny": 1.08}), ] CXP_PER_CATCH = 1 CXP_PER_BATTLE = 2 CXP_PER_TRADE = 1 CXP_PER_FORCE_SPAWN = 1 CXP_DAILY_CAP = 50 def get_chat_level_info(cxp: int): """CXP로 현재 레벨 + 혜택 조회.""" result = CHAT_LEVEL_THRESHOLDS[0] for threshold in CHAT_LEVEL_THRESHOLDS: if cxp >= threshold[1]: result = threshold else: break level, _, spawn_bonus, rarity_boosts = result # 다음 레벨까지 남은 CXP next_threshold = None for t in CHAT_LEVEL_THRESHOLDS: if t[1] > cxp: next_threshold = t[1] break return { "level": level, "spawn_bonus": spawn_bonus, "rarity_boosts": rarity_boosts, "next_cxp": next_threshold, # None이면 MAX }

1.5 queries.py 추가 함수

async def add_chat_cxp(chat_id: int, amount: int, action: str, user_id: int = None): """CXP 적립 (일일 상한 체크 포함). 레벨업 시 new_level 반환.""" pool = await get_db() # 일일 상한 체크 row = await pool.fetchrow( "SELECT cxp, chat_level, cxp_today FROM chat_rooms WHERE chat_id = $1", chat_id, ) if not row or row["cxp_today"] >= config.CXP_DAILY_CAP: return None # 상한 도달 actual = min(amount, config.CXP_DAILY_CAP - row["cxp_today"]) new_cxp = row["cxp"] + actual # 레벨 계산 info = config.get_chat_level_info(new_cxp) new_level = info["level"] leveled_up = new_level > row["chat_level"] await pool.execute( """UPDATE chat_rooms SET cxp = $1, chat_level = $2, cxp_today = cxp_today + $3 WHERE chat_id = $4""", new_cxp, new_level, actual, chat_id, ) # 로그 (선택적) await pool.execute( """INSERT INTO chat_cxp_log (chat_id, action, user_id, amount) VALUES ($1, $2, $3, $4)""", chat_id, action, user_id, actual, ) return new_level if leveled_up else None async def get_chat_level(chat_id: int): """채팅방 레벨 정보 조회.""" pool = await get_db() row = await pool.fetchrow( "SELECT cxp, chat_level, cxp_today FROM chat_rooms WHERE chat_id = $1", chat_id, ) if not row: return {"cxp": 0, "chat_level": 1, "cxp_today": 0} return dict(row) async def reset_daily_cxp(): """자정에 cxp_today 리셋.""" pool = await get_db() await pool.execute("UPDATE chat_rooms SET cxp_today = 0")
